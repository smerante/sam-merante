Standard datepicker

      Year: {{selectedDate | date: 'yyyy'}}
      Month: {{selectedDate | date: 'MM'}}
      Day: {{selectedDate | date: 'dd'}}
      Date: {{selectedDate | date: 'fullDate'}}
    }

Range datepicker

      Model: {	
      From:
        Year: {{selectedFromDate | date: 'yyyy'}}
        Month: {{selectedFromDate | date: 'MM'}}
        Day: {{selectedFromDate | date: 'dd'}}
        Date: {{selectedFromDate | date: 'fullDate'}}

      To:
        Year: {{selectedToDate | date: 'yyyy'}}
        Month: {{selectedToDate | date: 'MM'}}
        Day: {{selectedToDate | date: 'dd'}}
        Date: {{selectedToDate | date: 'fullDate'}}
      }

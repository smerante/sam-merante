var fact =[1,1];

function factorial(n)

function factorialFunc(n)
{
	if (n === 0) {
		return fact[1];
	  }
	  if((fact.length - 1) < n){
	  	  return fact[n] = factorialFunc(n-1)*n;
	  }
	  else
	  {
	  	   return fact[n];
	  }
}
